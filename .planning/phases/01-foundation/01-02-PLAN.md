export type Provider = "ollama" | "openrouter";

export interface ProviderConfig {
  provider: Provider;
  apiKey?: string;
  model?: string;
}

export interface AgentState {
  id: string;
  name: string;
  position: { x: number; y: number };
  activity: string;
  personality: string[];
}

export type WSMessageType = "agent_update" | "event" | "ping" | "pong" | "error";

export interface WSMessage {
  type: WSMessageType;
  payload: Record<string, unknown>;
  timestamp: number;
}

export interface SimulationStore {
  // State
  agents: Record<string, AgentState>;
  feed: WSMessage[];
  isConnected: boolean;
  isPaused: boolean;
  selectedAgentId: string | null;
  providerConfig: ProviderConfig | null;

  // Actions
  setAgents: (agents: Record<string, AgentState>) => void;
  appendFeed: (msg: WSMessage) => void;
  setConnected: (v: boolean) => void;
  setPaused: (v: boolean) => void;
  setSelectedAgent: (id: string | null) => void;
  setProviderConfig: (config: ProviderConfig) => void;
  reset: () => void;
}

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Browser localStorage → Zustand store | Provider config (including API key) lives in JS memory; no server-side storage in v1 |
| Zustand store → WebSocket messages | Inbound WS messages parsed and stored; malformed messages could corrupt store |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-01 | Information Disclosure | Zustand store (api_key in providerConfig) | mitigate | In Phase 1, providerConfig is set to null until Plan 03 wires it. Key never logged to console. ActivityFeed never renders raw providerConfig. |
| T-02-02 | Tampering | useWebSocket WSMessage parsing | mitigate | Wrap JSON.parse in try/catch in useWebSocket; malformed messages are logged and discarded, not appended to feed. |
| T-02-03 | Denial of Service | WebSocket reconnect loop | mitigate | Reconnect uses 3-second delay. Cap reconnect attempts at 10 before showing persistent error state (avoids infinite tight loop). |
</threat_model>
