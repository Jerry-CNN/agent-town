   ```python
   import asyncio, json, pytest, time
   from httpx import AsyncClient, ASGITransport
   from backend.main import app

   @pytest.mark.asyncio
   async def test_health():
       async with AsyncClient(transport=ASGITransport(app=app), base_url="http://test") as client:
           resp = await client.get("/health")
       assert resp.status_code == 200
       assert resp.json() == {"status": "ok"}

   @pytest.mark.asyncio
   async def test_ollama_status_keys(mock_ollama_unreachable):
       async with AsyncClient(transport=ASGITransport(app=app), base_url="http://test") as client:
           resp = await client.get("/api/ollama/status")
       assert resp.status_code == 200
       data = resp.json()
       for key in ("available", "models", "has_default_model", "error"):
           assert key in data, f"Missing key: {key}"
       assert data["available"] is False

   @pytest.mark.asyncio
   async def test_ollama_status_available(mock_ollama_available):
       async with AsyncClient(transport=ASGITransport(app=app), base_url="http://test") as client:
           resp = await client.get("/api/ollama/status")
       assert resp.status_code == 200
       data = resp.json()
       assert data["available"] is True
       assert data["has_default_model"] is True

   @pytest.mark.asyncio
   async def test_llm_test_endpoint(monkeypatch):
       from backend.llm import gateway
       from backend.models.messages import TestResponse  # or define inline

       async def mock_llm_call(*args, **kwargs):
           # Simulate a successful Pydantic-validated response
           class FakeResult:
               text = "Hello"
           return FakeResult()

       monkeypatch.setattr(gateway, "llm_call", mock_llm_call)
       async with AsyncClient(transport=ASGITransport(app=app), base_url="http://test") as client:
           resp = await client.get("/api/llm/test?prompt=hello")
       assert resp.status_code == 200
       assert "response" in resp.json()

   @pytest.mark.asyncio
   async def test_websocket_handshake():
       from starlette.testclient import TestClient
       with TestClient(app) as client:
           with client.websocket_connect("/ws/test-session") as ws:
               data = ws.receive_json()
       assert data["type"] == "sim_connected"
       assert data["session_id"] == "test-session"

   @pytest.mark.asyncio
   async def test_concurrent_agent_steps():
       """INF-02: all 10 agents must complete in parallel, not sequential."""
       async def fake_step(i: int) -> float:
           await asyncio.sleep(0.1)
           return time.monotonic()

       start = time.monotonic()
       async with asyncio.TaskGroup() as tg:
           tasks = [tg.create_task(fake_step(i)) for i in range(10)]
       elapsed = time.monotonic() - start
       assert elapsed < 0.5, f"Steps ran sequentially: {elapsed:.2f}s (expected <0.5s)"
   ```
