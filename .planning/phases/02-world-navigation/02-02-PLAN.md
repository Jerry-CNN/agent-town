    ```python
    class AgentScratch(BaseModel):
        """Agent personality and background -- the 'scratch pad' from the reference."""
        age: int
        innate: str          # personality traits, comma-separated e.g. "warm, creative, curious"
        learned: str         # background paragraph
        lifestyle: str       # sleep/wake patterns
        daily_plan: str      # routine template for LLM to fill in Phase 3

    class AgentSpatial(BaseModel):
        """Agent's spatial knowledge -- known locations and their structure."""
        address: dict[str, list[str]]   # named addresses e.g. {"living_area": ["agent-town", "home-alice", "bedroom"]}
        tree: dict                       # spatial knowledge tree (nested dict of known locations)

    class AgentConfig(BaseModel):
        """Complete agent configuration loaded from JSON file."""
        name: str
        coord: tuple[int, int]           # spawn position (x, y)
        currently: str                   # current situation summary (seed for Phase 3)
        scratch: AgentScratch
        spatial: AgentSpatial
    ```

    ```python
    import json
    from pathlib import Path
    from backend.schemas import AgentConfig

    AGENTS_DIR = Path(__file__).parent.parent / "data" / "agents"

    def load_all_agents() -> list[AgentConfig]:
        """Load all agent JSON configs from data/agents/, validate with Pydantic."""
        configs = []
        for f in sorted(AGENTS_DIR.glob("*.json")):
            raw = json.loads(f.read_text())
            configs.append(AgentConfig.model_validate(raw))
        return configs
    ```
